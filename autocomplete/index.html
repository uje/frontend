AutoComplete示例

调用方式 
new AutoComplete({
    el             : '#keyword'                // 文本框（这里使用document.querySelector获取）
    interval       : 500,                      // 触发ajax间隔
    containerClass : 'ac-container',           // 外层容器className
    inputClass     : 'ac-input',               // 输入框className
    listContainerClass: 'ac-list-container',   // 列表容器ClassName
    listClass      : 'ac-list',                // 列表className
    itemClass      : 'ac-item',                // 数据条目className
    oddClass       : 'ac-odd',
    evenClass      : 'ac-even',
    selectedClass  : 'ac-selected',            // 选中条目ClassName
    buildList: function(){
        return '<ul></ul>';
    },
    buildItem: function(item){
        return '<li><a href="javascript:;">' + item +'</a></li>';
    },
    buildEmptyTemplate: function(){
        return '<li class="ac-empty">没有数据.</li>';
    },
    formatData: function(data){
        return data;
    },
    formatSelectedData: function(origialData){
        return origialData;
    },
    created: emptyFn,  // 控件创建后触发
    selected: emptyFn  // 选择后触发
})
